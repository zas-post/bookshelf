# System, applies to entire machine and all users
$ git config --system --list
$ git config --system --edit

# User defined
$ git config --global --list
$ git config --global --edit

git config --list

git config --list --show-origin

> git config --global user.name ""    - добавление пользователя

> git config --global user.email ""   - добавление email

> git config -l                       - просмотр конфига

# Работа с Local Repository

1. создаём локальную папку на компе
mkdir myproject

cd myproject

git init .

можно создать сразу 
git init myproject

git status

echo "www" > file-1.txt  создание файла

git add . или (*)

git commit -m "Добавление коментария, ver 1.0"


git log
git log -1  колличество коммитов (посследний коммит)
git log -l -p  покажет что было сделано в коммит

git checkout -- file1.txt   - вернуть файл на начало, до стейжинга

git diff --stage   - какие изменения закоммитятся
